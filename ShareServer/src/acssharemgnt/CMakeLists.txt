# 包含prep.cmake(定义编译时的选项)
INCLUDE($ENV{CMAKE}/common/prep.cmake)

# 设置版本号，和向下兼容性
CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
CMAKE_POLICY(VERSION 2.8)

# project name: ACSSHAREMGNT
PROJECT(ACSSHAREMGNT)

SET(NEED_CFL TRUE)
SET(NEED_XPCOM TRUE)
SET(NEED_NCUTIL TRUE)
SET(NEED_SYSWRAP TRUE)
SET(NEED_EDATAAPI TRUE)
SET(NEED_EDBACCESS TRUE)
SET(ENABLE_C++11 TRUE)
SET(NEED_ETHRIFT TRUE)
SET(NEED_THRIFTAPI TRUE)
SET(NEED_BRPC TRUE)

SET(ERRGEN_FILES $ENV{SHARESERVER}/src/acssharemgnt/err/acssharemgnterr.csv)

ADD_DEFINITIONS("-I$ENV{SHARESERVER}")

# 包含 ShareServer目录，主要是依赖acsdb
INCLUDE_DIRECTORIES($ENV{SHARESERVER}/src)

# target name: acssharemgnt
SET(TARGET_NAME acssharemgnt)

# TARGET_TYPE
SET(TARGET_TYPE COMPONENT)

INCLUDE($ENV{CMAKE}/common/target.cmake)
