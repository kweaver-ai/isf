#包含prep.cmake(定义编译时的选项)

#设置版本号，和向下兼容性
CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
CMAKE_POLICY(VERSION 2.8)

# project name
PROJECT(ACS_UTIL)

SET(NEED_XPCOM TRUE)
SET(NEED_CFL TRUE)
SET(NEED_SYSWRAP TRUE)
SET(NEED_NCUTIL TRUE)
SET(NEED_BOOST TRUE)
SET(NEED_ETHRIFT TRUE)
SET(NEED_BOOST_DATE_TIME TRUE)
SET(NEED_BOOST_REGEX TRUE)
SET(NEED_EDBACCESS TRUE)
SET(NEED_OPENSSL TRUE)
SET(ENABLE_C++11 TRUE)
SET(NEED_THRIFTAPI TRUE)

LINK_DIRECTORIES (/usr/lib64/mysql/)
LINK_LIBRARIES (mysqlclient)

INCLUDE_DIRECTORIES($ENV{SHARESERVER}/src)

# target name
SET(TARGET_NAME acs_util)

# target type
SET(TARGET_TYPE RUNTIME)

ADD_DEFINITIONS("-DHAVE_INTTYPES_H")
ADD_DEFINITIONS("-DHAVE_NETINET_IN_H")

INCLUDE($ENV{CMAKE}/common/target.cmake)
